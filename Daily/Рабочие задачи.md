## Таблица задач с цветами, прогрессом и сортировкой P1 сверху

```dataviewjs
const today = dv.date("today");

// Берём задачи из текущего файла
let tasks = dv.current().file.tasks.where(t => t.text);

// Сортировка по приоритету P1-P4
const priorityOrder = { P1:1, P2:2, P3:3, P4:4 };
tasks.sort((a,b) => priorityOrder[a.priority ?? "P4"] - priorityOrder[b.priority ?? "P4"]);

// Функция прогресс-бара
function progressBar(value){
    value = value ?? 0;
    return `<div style="width:100px;height:10px;background:#444;border-radius:6px;margin:auto;">
              <div style="width:${value}%;background:#4caf50;height:10px;border-radius:6px;"></div>
            </div>
            <div style="text-align:center;font-size:0.8em;">${value}%</div>`;
}

// --- проверка подзадач и прогресса ---
tasks.forEach(t => {
    // Получаем плановую дату из текста задачи [YYYY-MM-DD]
    let dueMatch = t.text.match(/\[(\d{4}-\d{2}-\d{2})\]/);
    t.dueParsed = dueMatch ? dv.date(dueMatch[1]) : null;

    // Если задача закрыта, прогресс = 100% и дата выполнения = сегодня
    if(t.completed){
        t.progress = 100;
        t.status = "выполнено";
        if(!t.completedDate) t.completedDate = dv.date("today");
    }

    // Проверяем подзадачи
    if(t.children && t.children.length > 0){
        const total = t.children.length;
        const done = t.children.filter(c => c.completed).length;

        t.progress = Math.round((done/total)*100);

        // Если все подзадачи выполнены → закрываем основную задачу
        if(done === total){
            t.completed = true;
            t.progress = 100;
            t.status = "выполнено";
            t.completedDate = dv.date("today");
        }
    }
});

// --- строим таблицу вручную ---
let table = dv.el("table","",{attr:{style:"border-collapse: collapse; width: 100%; font-family: sans-serif;"}});
let thead = table.createTHead();
let row = thead.insertRow();
["Задача","Приоритет","Статус","Плановая дата","Реальная дата","Срок OK ⭐️","Прогресс","Причина"].forEach(h=>{
    let th = document.createElement("th");
    th.innerText = h;
    th.style.border="1px solid #999";
    th.style.padding="6px 4px";
    th.style.background="#eee";
    th.style.fontSize="0.9em";
    th.style.textAlign="center";
    row.appendChild(th);
});

let tbody = table.createTBody();
tasks.forEach(t=>{
    let tr = tbody.insertRow();

    // Цвет строки по статусу
    let color="";
    if(t.completed) color="#2e7d32";          // зеленый
    else if(t.dueParsed && t.dueParsed < today) color="#c62828"; // красный
    else if(t.status=="в работе") color="#f9a825";              // желтый
    else if(t.status=="приостановлено") color="#616161";        // серый

    // ⭐️ если выполнено в срок
    let star="";
    if(t.completed && t.completedDate && t.dueParsed){
        if(t.completedDate <= t.dueParsed) star="⭐️";
    }

    let cells=[
        t.text.split("\n")[0],
        t.priority ?? "",
        t.status ?? "",
        t.dueParsed ? t.dueParsed.toFormat("yyyy-MM-dd") : "",
        t.completedDate ? t.completedDate.toFormat("yyyy-MM-dd") : "",
        star,
        progressBar(t.progress),
        t.reason ?? ""
    ];

    cells.forEach((c,i)=>{
        let td=tr.insertCell();
        td.style.border="1px solid #999";
        td.style.padding="4px";
        td.style.fontSize="0.85em";
        td.style.textAlign=(i===0)?"left":"center";
        td.style.verticalAlign="middle";
        td.style.maxWidth="200px";
        td.style.overflow="hidden";
        td.style.whiteSpace="nowrap";
        td.style.textOverflow="ellipsis";
        if(i!==6) td.style.backgroundColor=color;
        td.innerHTML=c;
    });
});
```
